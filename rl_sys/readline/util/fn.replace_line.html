pub fn replace_line(text: &str, clear_undo: bool) -> Result<(), ReadlineError>

Replace the contents of rl_line_buffer with text. The point and mark are preserved, if possible. If clear_undo is true, the undo list associated with the current line is cleared.

Examples

use rl_sys::readline::util;

util::init();

assert!(util::replace_line("replacement", false).is_ok());
